第一题

function setName(){
name = “张三”
} setName()
console.log(‘setName’)
答案： 张三

第二题

//考点：1.变量声明提升 2.变量搜索机制
var a = 1
function test(){
console.log(a)
var a = 1;
} test();
答案：undefined

第三题

var b = 2;
function text2(){
window.b=3;
console.log();
} test2;
答案：3

第四题

c=5; //声明一个全局变量c
function test3(){
window.c=3;
console.log(3) 答案：undefined,原因：由于此时得c是一个局部变量c，并没有被赋值
var c;
console.log(window.c); //答案：3,原因：这里得c就是一个全局变量c
} test3();

第五题

var arr = [];
arr[0] = ‘a’;
arr[1] = ‘b’;
arr[10] = ‘c’;
alert(arr.length)//答案：11
console.log(arr[5]) //答案：undefined

第六题

var a = 1;
console.log(a++); 答案：1
console.log(++a); 答案：3

第七题

console.log(null==undefined) //答案：true
console.log(“1”==1) //答案：true,因为会将数字1先转换为字符串1
console.log(“1”===1) //答案：false,因为数据类型不一致
